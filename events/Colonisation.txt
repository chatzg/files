country_event = {
	id = colonisation.4
	title = "Our North American Colonies merge"
	desc = "The newly founded colony in North America has decided to merge with [USA.GetNameDEF]"
	picture = GFX_report_event_generic_sign_treaty2

	
	fire_only_once = no 
	is_triggered_only = no
	trigger = {
		any_country = {
			
			original_tag = AME
			is_subject_of = ROOT
			
		}
		USA = {
			is_subject_of = ROOT
		}
	}
	mean_time_to_happen = {
		days = 2
	}
	option = {
		name = "Ok, let's hope they don't rebel"
		
			random_country = {
				limit = {
					is_subject_of = ROOT
					original_tag = AME
				}
				every_owned_state = {
					limit = {
						is_core_of = PREV
					}
					add_core_of = USA
				}
				USA = {
					annex_country = {
						target = PREV
						transfer_troops = yes
					}
				}
			}
		
	}
}
